function recalcScores() {

  state.scores = {};

  state.players.forEach(p => { state.scores[p] = { draft: 0, prediction: 0 }; });

  // +5 per elimination survived per drafted player

  state.eliminations.forEach(elim => {

    state.players.forEach(player => {

      (state.draftPicks[player] || []).forEach(cid => {

        if (cid !== elim.contestantId) state.scores[player].draft += 5;

      });

    });

  });

  // Winner prediction: +50 if correct

  if (state.soleSurvivor) {

    state.players.forEach(player => {

      if (state.winnerPredictions[player] === state.soleSurvivor) {

        state.scores[player].prediction += 50;

      }

    });

  }

  // Idol predictions: +15 each if correct

  ['idol1', 'idol2'].forEach(idol => {

    const outcome = state.idolOutcomes?.[idol];

    if (!outcome) return;

    state.players.forEach(player => {

      if (state.idolPredictions?.[player]?.[idol] === outcome) {

        state.scores[player].prediction += 15;

      }

    });

  });

  saveState();

}

    const draftIds = state.draftPicks[player] || [];

    if (draftIds.includes(cid)) { toast('Sole Survivor cannot be one of your draft players!'); return; }

    state.winnerPredictions[player] = cid;

  }

  // Idols

  state.idolPredictions[player] = state.idolPredictions[player] || {};

  if (!state.idolPredictions[player].idol1) {

    if (!pending.idol1) { toast('Select an outcome for Idol 1'); return; }

    state.idolPredictions[player].idol1 = pending.idol1;

  }

  if (!state.idolPredictions[player].idol2) {

    if (!pending.idol2) { toast('Select an outcome for Idol 2'); return; }

    state.idolPredictions[player].idol2 = pending.idol2;

  }

  saveState();

  renderPredictions();

  toast(`${player}'s predictions locked! 🔮`);

}

function addPlayer() {

  const name = document.getElementById('new-player-name').value.trim();

  if (!name) { toast('Enter a name first'); return; }

  if (state.players.includes(name)) { toast('Player already exists'); return; }

  state.players.push(name);

  state.scores[name] = { draft: 0, weekly: 0 };

  saveState();

  document.getElementById('new-player-name').value = '';

  document.getElementById('player-count').textContent = state.players.length;

  renderDraftAssignments();

  toast(`Added ${name} to the pool`);

}

function recordElimination() {

  const week = parseInt(document.getElementById('elim-week').value);

  const cid = parseInt(document.getElementById('elim-contestant').value);

  const idol = document.getElementById('elim-idol').value;

  const idolPlayed = idol === 'yes';

  const isMedevac = idol === 'medevac';

  if (isNaN(week) || isNaN(cid) || !cid) { toast('Fill in all fields'); return; }

  if (state.eliminations.find(e => e.week === week)) { toast('Week already recorded'); return; }

  state.eliminations.push({ week, contestantId: cid, idolPlayed, isMedevac });

  const c = getContestant(cid);

  if (c) c.alive = false;

  recalcScores();

  saveState();

  populateElimSelect();

  toast(`${c?.name || 'Contestant'} voted out in Week ${week}`);

}

function renderDraftAssignments() {

  const container = document.getElementById('draft-assignments');

  if (state.players.length === 0) {

    container.innerHTML = `<p style="color:var(--muted);font-size:0.75rem">No players yet.</p>`;

    return;

  }

  container.innerHTML = state.players.map(player => {

    const picked = state.draftPicks[player] || [];

    const aliveContestants = state.cast.filter(c => c.alive);

    const unowned = state.cast.filter(c => {

      const ownerExists = state.players.some(p => (state.draftPicks[p]||[]).includes(c.id));

      return !ownerExists || (state.draftPicks[player]||[]).includes(c.id);

    });

    const options = state.cast.map(c => {

      const ownerExists = state.players.some(p => p !== player && (state.draftPicks[p]||[]).includes(c.id));

      return `<option value="${c.id}" ${picked.includes(c.id)?'selected':''} ${ownerExists?'disabled':''}>

        ${c.name}${ownerExists?' (taken)':''}${!c.alive?' ✗':''}

      </option>`;

    }).join('');

    return `<div style="margin-bottom:1rem">

      <label style="font-size:0.7rem;font-family:'Cinzel',serif;color:var(--muted);letter-spacing:0.1em;display:block;margin-bottom:0.4rem">${player}</label>

      <select multiple style="width:100%;background:var(--surface);border:1px solid rgba(255,255,255,0.1);color:var(--text);padding:0.4rem;font-size:0.8rem;border-radius:2px;height:80px" onchange="updateDraftPick('${player}', this)">

        ${options}

      </select>

    </div>`;

  }).join('');

}

function updateDraftPick(player, sel) {

  const selected = Array.from(sel.selectedOptions).map(o => parseInt(o.value));

  state.draftPicks[player] = selected;

  saveState();

  recalcScores();

  toast(`Draft picks updated for ${player}`);

}

function recordAdvantage() {

}
